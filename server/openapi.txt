  version: 1.0.0

servers:
  - url: /

paths:
  /status:
    get:
      tags:
        - private
      summary: Kubernetes readiness and liveness probe endpoint
      responses:
        '200':
          description: successful operation

  /api-docs:
    get:
      tags:
        - public
      summary: List of APIs provided by the service
      responses:
        '200':
          description: OK

  /deploy:
    post:
      tags:
        - public
      summary: deploy the application
      requestBody:
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/DeployPayload'
            encoding:
              payload:
                contentType: application/json
      responses:
        '200':
          description: OK

  /register/agent:
    post:
      tags:
        - public
      summary: Register agent
      requestBody:
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/AgentRequest'
      responses:
        '200':
          description: OK
    put:
      tags:
        - public
      summary: Register agent
      requestBody:
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/AgentRequest'
      responses:
        '200':
          description: OK
    get:
      tags:
        - public
      summary: Register agent
      responses:
        '200':
          description: OK
          content:
            application/json:
              schema:
                type: array
                items:
                  $ref: '#/components/schemas/AgentRequest'

components:
  schemas:
    AgentRequest:
      title: Agent information
      type: object
      properties:
        customer_id:
          type: string
        endpoint:
          type: string
      required:
        - customer_id
        - endpoint

    DeployPayload:
      title: Deploy payload
      type: object
      properties:
        operation:
          type: string
          description: Operation
        plugin:
          type: string
          description: plugin
        workertype:
          type: string
          description: workertype
        payload:
          type: string
          description: payload
        
        payload:
          $ref: '#/components/schemas/DeployRequestPayload'
      required:
        - operation
        - payload

    DeployRequestPayload:
      title: Deploy operation request payload
      type: object
      properties:
        plugin_name:
          type: string
          description: Plugin name for the operation
        action:
          type: string
          description: Action to be performed
        data:
          type: object
          description: Data for the action
      required:
        - plugin_name
        - action
        - data
